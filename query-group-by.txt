QUERY 1 

CONTARE QUANTI ISCRITTI CI SONO STATI OGNI ANNO 

SELECT COUNT(id) 
FROM `students`
WHERE YEAR(`enrolment_date`) 
GROUP BY YEAR(`enrolment_date`) 

QUERY 2 

CONTARE GLI INSEGNANTI CHE HANNO L'UFFICIO NELLO STESSO EDIFICIO 

SELECT COUNT(id) 
FROM `teachers`
WHERE`office_number`
GROUP BY `office_number`

QUERY 3 

CALCOLARE LA MEDIA DEI VOTI DI OGNI APPELLO 

SELECT AVG(vote) 
FROM `exam_student`
WHERE `exam_id`
GROUP BY `exam_id`

QUERY 4 

CONTAREA QUANTI CORSI DI LAUREA CI SONO PER OGNI DIPARTIMENTO 

SELECT COUNT(id) 
FROM `degrees`
WHERE `department_id`
GROUP BY `department_id`
