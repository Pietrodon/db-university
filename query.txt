QUERY 1

SELEZIONARE TUTTI GLI STUDENTI NATI NEL 1990

SELECT * 
FROM `students` 
WHERE `date_of_birth` LIKE '1990%'


QUERY 2 

SELEZIONARE TUTTI I CORSI CHE VALGONO PIU DI 10 PUNTI 

SELECT * 
FROM `courses` 
WHERE `cfu` > 10

QUERY 4 
 
SELEZIONARE TUTTI I CORSI DEL PRIMO SEMESTRE DI QUALSIASI CORSO DI LAUREA 

SELECT * 
FROM `courses` 
WHERE `period` = "I semestre" AND `year` ="1"

QUERY 5 

SELEZIONARE TUTTI GLI APPELLI CHE AVVENGONO NEL POMERIGGIO (DOPO LE 14) DEL 20/06/2020

SELECT * 
FROM `exams` 
WHERE `date` = "2020-06-20" AND `hour` LIKE "14%"
